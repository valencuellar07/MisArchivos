using System;
public class Program{
  public static void Main() {
        // Display the number of command line arguments.
        Console.WriteLine(CountBits(1234));
    }
  
  public static int CountBits(int n){
	const int POWER_BASE=2;
    bool stop=false;
    double power=0;
	int counter=1;
	do{
      
      power=Math.Pow(POWER_BASE, counter); 
      counter++;
      if(power>n){
        stop=true;
      }
    }
	while (stop==false);
    int result=ValidationBits(counter,n);
    return result;
  }
  
  public static int ValidationBits(int counter, int n){
    const int POWER_BASE=2;
    double power=0;
    double minus=0;
    int countOnes=0;
	double plus=0;
    for (int index=counter-1; index>-1; index--){
	  Console.WriteLine("bits"+index);
      if(index==counter-1){minus=n;}
      if(minus<(power=Math.Pow(POWER_BASE, index))){
        continue;
      }
      if(minus>=(power=Math.Pow(POWER_BASE, index))){
		  power=Math.Pow(POWER_BASE, index);
		  plus+=power;
		  Console.WriteLine("power"+power);
		  minus=n-plus;
		  countOnes+=1;
	  }
      if(plus==n){
        break;
      }   
    }
     return countOnes;
  }

}
